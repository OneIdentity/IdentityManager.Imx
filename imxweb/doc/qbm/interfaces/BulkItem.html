import { TypedEntity } from 'imx-qbm-dbts';
import { ColumnDependentReference } from '../../cdr/column-dependent-reference.interface';
import { EntitySelect } from '../../entity/entity-select/entity-select.interface';

export enum BulkItemStatus {
  unknown,
  saved,
  skipped,
  error,
  valid
}

export interface BulkItem {
  entity: TypedEntity;
  properties: ColumnDependentReference[];
  additionalInfo?: string;
  status: BulkItemStatus;
  customTitle?: string;
  description?: string;
  customSelectProperties?: EntitySelect[];
  valid?: boolean;
  mandatory?: boolean;
  readonly?: boolean;
}
